Instacart Dashboard

loading and tidying the instacart dataset

# loading dataset
data("instacart")

# tidying dataset 
instacart_df =
  instacart %>% 
  janitor::clean_names() %>% 
  filter(
    department %in% c("frozen"), 
    order_dow == 6
  )

  layout(title = "Items ordered in each aisle on day 6",
